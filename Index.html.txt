// ─── MAIN APP ────────────────────────────────────────────────────────────────const WorldBenchRegister = () => {  const [benches, setBenches] = useState([]);  const [showForm, setShowForm] = useState(false);  const [toDelete, setToDelete] = useState(null);  const [search, setSearch] = useState('');  const [filterTag, setFilterTag] = useState('');  const [sort, setSort] = useState('recent');
  useEffect(() => {    const saved = localStorage.getItem('world-benches-v2');    if (saved) setBenches(JSON.parse(saved));  }, []);
  const persist = (list) => {    setBenches(list);    localStorage.setItem('world-benches-v2', JSON.stringify(list));  };
  const handleSave = (bench) => {    persist([bench, ...benches]);    setShowForm(false);  };
  const handleDelete = (bench) => setToDelete(bench);  const confirmDelete = () => {    persist(benches.filter(b => b.id !== toDelete.id));    setToDelete(null);  };

  const hasMapped = filtered.some(b => b.lat && b.lng);
  return (    <div style={{ minHeight: '100vh' }}>      <header style={{        background: 'var(--forest)', color: 'var(--cream)',        padding: '1.2rem 1.5rem', display: 'flex', justifyContent: 'space-between',        alignItems: 'center', position: 'sticky', top: 0, zIndex: 50,        boxShadow: '0 4px 30px rgba(30,58,47,0.3)'      }}>        <div>          <h1 style={{ fontFamily: 'Playfair Display', fontSize: '1.3rem' }}> World Bench Register</h1>          <p style={{ fontSize: '0.55rem', letterSpacing: '0.15em', textTransform: 'uppercase', opacity: 0.6 }}>L'inventaire mondial des assises</p>        </div>        <div style={{ display: 'flex', alignItems: 'center', gap: 12 }}>          <span style={{ fontSize: '0.8rem', opacity: 0.7 }}>{benches.length} banc{benches.length !== 1 ? 's' : ''}</span>          <button className="btn-primary" style={{ padding: '8px 18px', fontSize: '0.75rem' }}            onClick={() => setShowForm(true)}>+ Ajouter</button>        </div>      </header>      <div style={{ height: 4, background: 'linear-gradient(90deg, var(--forest) 0%, var(--amber) 50%, var(--forest-light) 100%)' }} />      <main style={{ maxWidth: 1100, margin: '0 auto', padding: '2rem 1.2rem' }}>        <FilterBar search={search} setSearch={setSearch}          filterTag={filterTag} setFilterTag={setFilterTag}          sort={sort} setSort={setSort} />        {hasMapped && (          <div style={{ marginBottom: 40 }}>            <p className="section-label" style={{ marginBottom: 14 }}> Carte des spots</p>            <MapView benches={filtered} />          </div>        )}        {filtered.length === 0 ? (          <div style={{ textAlign: 'center', padding: '4rem 2rem', border: '2px dashed var(--cream-dark)', borderRadius: 28 }}>            <p style={{ fontSize: '3rem', marginBottom: 12 }}> </p>            <p style={{ color: 'var(--stone)', fontSize: '1.05rem', fontWeight: 500 }}>              {benches.length === 0 ? 'Aucun banc répertorié.' : 'Aucun résultat.'}            </p>            <p style={{ color: 'var(--stone-light)', fontSize: '0.85rem', marginTop: 6 }}>              {benches.length === 0 ? 'Soyez le premier explorateur à contribuer !' : 'Essayez d\'autres filtres.'}            </p>            {benches.length === 0 && (              <button className="btn-primary" style={{ marginTop: 24 }} onClick={() => setShowForm(true)}>                Ajouter le premier banc →              </button>            )}          </div>        ) : (          <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill, minmax(300px, 1fr))', gap: 24 }}>            {filtered.map(b => <BenchCard key={b.id} bench={b} onDelete={handleDelete} />)}          </div>        )}      </main>      <footer style={{ textAlign: 'center', padding: '2rem', color: 'var(--stone-light)', fontSize: '0.7rem', letterSpacing: '0.1em' }}>        WORLD BENCH REGISTER — L'inventaire mondial des assises      </footer>      {showForm && <BenchForm onSave={handleSave} onClose={() => setShowForm(false)} />}      {toDelete && <ConfirmDialog bench={toDelete} onConfirm={confirmDelete} onCancel={() => setToDelete(null)} />}    </div>  );};
ReactDOM.createRoot(document.getElementById('root')).render(<WorldBenchRegister />);</script></body></html>
